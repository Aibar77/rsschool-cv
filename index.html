Assignment: find a unique number from array

            
  function findUniq(arr) {
  let arr1 = [];
  let arr2 = [];
  for (let num of arr) {
    if (arr1.length === 0) {
      arr1.push(num);
    } else {
      if (num === arr1[0]) {
        arr1.push(num);
      } else {
        arr2.push(num);
      }
    }
  }
  return arr1.length > arr2.length ? arr2[0] : arr1[0];
}
